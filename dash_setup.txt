# Run the following commands to install Dash on your machine.

sudo pip install --upgrade pip # Upgrade pip
sudo /usr/local/bin/pip install dash==0.22.0  # The core dash backend
sudo /usr/local/bin/pip install dash-renderer==0.13.0  # The dash front-end
sudo /usr/local/bin/pip install dash-html-components==0.11.0  # HTML components
sudo /usr/local/bin/pip install dash-core-components==0.26.0  # Supercharged components
sudo /usr/local/bin/pip install plotly --upgrade  # Plotly graphing library used in examples
sudo /usr/local/bin/pip install dash-table-experiments # Display Tables in Dash
sudo /usr/local/bin/pip install pandas # Install Pandas
sudo /usr/local/bin/pip install cassandra-driver # Install Cassandra Driver
